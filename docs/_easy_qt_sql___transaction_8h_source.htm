77  static int range(QueryResult &res, int start, int count, Func&& f)
78  {
79  Q_ASSERT(start >= 0);
80  Q_ASSERT(count >= 0);
81 
82  int rowCount = 0;
83 
84  if (res.isActive())
85  {
86  //skip items
87  for (int i = 0; i < start; ++i)
88  {
89  if (!res.next())
90  return rowCount;
91  }
92 
93  for (int i = 0; i < count && res.next(); ++i)
94  {
95  f(res);
96  rowCount++;
97  }
98  }
99 
100  return rowCount;
101  }
